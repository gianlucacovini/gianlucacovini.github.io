\cvsection{Research}

\cvsubsection{Research Interests}
I am interested in \textbf{optimal transport}, especially in the use of tools from \emph{calculus of variations, gradient flows and differential geometry} to study problems arising in statistics, optimization, and machine learning.

\cvsubsection{Current Projects}
\compactevent{Sinkhorn Propagation on Graphs}{with Hugo Lavenant, Bocconi University}{Ongoing}{}
We study \textbf{variational propagation problems on graphs} involving Sinkhorn divergences, using \emph{variational methods}, \emph{gradient-flow formulations}, and numerical algorithms.

\smallskip
\compactevent{Bayes Rule via Entropic Approximation of the Knothe--Rosenblatt Rearrangement}{with Carlo Ciliberto and Giulia Luise}{Ongoing}{}
We study \textbf{entropic approximations of the Knothe--Rosenblatt rearrangement} for \emph{posterior sampling in Bayesian inference}.

\cvsubsection{Publication}
\textbf{G. Covini}, D. Antipov, and C. Doerr. \textit{Enhancing Parameter Control Policies with State Information}. Proceedings of FOGA 2025. \href{https://arxiv.org/abs/2507.08368}{arXiv:2507.08368}.

\cvsubsection{Research Software}
